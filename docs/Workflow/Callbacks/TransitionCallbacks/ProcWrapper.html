def wrapper
  arguments = [
    name_arguments_string,
    rest_param_string,
    kw_arguments_string,
    keyrest_string,
    procedure_string].compact.join(', ')

  raw_proc = self.raw_proc
  str = build_proc("target.instance_exec(#{arguments})")
  eval(str)
end
